defined_text = {
	name = GetLeninComemmorationAccepted
	text = {
		trigger = {
			any_country = {
				has_country_flag = lenin_comemmoration_accepted
			}
		}
		localization_key = lenin_death_accepted
	}
	text = {
		trigger = {
			all_country = {
				NOT = { has_country_flag = lenin_comemmoration_accepted }
			}
		}
		localization_key = lenin_death_accepted_none
	}
}
