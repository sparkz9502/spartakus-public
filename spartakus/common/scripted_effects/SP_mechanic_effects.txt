SP_soc_cap_embargo_effect = {
	if = {
		limit = {
			OR = {
				OR = {
					original_tag = ENG
					is_in_faction_with = ENG
				}
				OR = {
					original_tag = FRA
					is_in_faction_with = FRA
				}
				OR = {
					original_tag = JAP
					is_subject_of = JAP
				}
				OR = {
					original_tag = ITA
					is_subject_of = ITA
				}
				has_country_flag = SP_embargoed_socialists
			}
		}
		every_country = {
			limit = {
				has_socialist_government = yes
			}
			add_opinion_modifier = {
				target = PREV
				modifier = SP_soc_cap_embargo
			}
			reverse_add_opinion_modifier = {
				target = PREV
				modifier = SP_soc_cap_embargo
			}
		}
	}
}

SP_remove_ideology_opinion = {
	every_country = {
		remove_opinion_modifier = {
			target = ROOT
			modifier = SP_similar_ideology
		}
		ROOT = {
			remove_opinion_modifier = {
				target = PREV
				modifier = SP_similar_ideology
			}
		}
		remove_opinion_modifier = {
			target = ROOT
			modifier = SP_same_ideology
		}
		ROOT = {
			remove_opinion_modifier = {
				target = PREV
				modifier = SP_same_ideology
			}
		}
		remove_opinion_modifier = {
			target = ROOT
			modifier = SP_capitalist_v_socialist_ideology
		}
		ROOT = {
			remove_opinion_modifier = {
				target = PREV
				modifier = SP_capitalist_v_socialist_ideology
			}
		}
		remove_opinion_modifier = {
			target = ROOT
			modifier = SP_liberal_v_right_ideology
		}
		ROOT = {
			remove_opinion_modifier = {
				target = PREV
				modifier = SP_liberal_v_right_ideology
			}
		}
	}
}
