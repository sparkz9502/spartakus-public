defined_text = {
    name = GetSERLiteracyVariables
    text = {
        trigger = {
            check_variable = {
                var = SER.literacy_rate
                value = 0
                compare = greater_than_or_equals
            }
            check_variable = {
                var = SER.literacy_rate
                value = 50
                compare = less_than_or_equals
            }
        }
        localization_key = SER_literacy_rate_low_variable
    }
    text = {
        trigger = {
            check_variable = {
                var = SER.literacy_rate
                value = 51
                compare = greater_than_or_equals
            }
            check_variable = {
                var = SER.literacy_rate
                value = 75
                compare = less_than_or_equals
            }
        }
        localization_key = SER_literacy_rate_medium_variable
    }
	
	text = {
        trigger = {
            check_variable = {
                var = SER.literacy_rate
                value = 76
                compare = greater_than_or_equals
            }
            check_variable = {
                var = SER.literacy_rate
                value = 100
                compare = less_than_or_equals
            }
        }
        localization_key = SER_literacy_rate_high_variable
    }
}
